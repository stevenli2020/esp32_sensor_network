// Every 5 mins of average (working on phpadmin query)
SELECT AVG(SENSOR_DATA), CONVERT((MIN(TIME) DIV 500) * 500 + 230, TIME) as time FROM EVENTS WHERE TIME BETWEEN NOW() - INTERVAL 1 DAY AND NOW() GROUP BY TIME DIV 500;

// more accurate
SELECT AVG(SENSOR_DATA) as average, DATE_FORMAT(MIN(TIME), '%Y-%m-%d %H:%i:00') AS TIME
   FROM EVENTS WHERE TIME BETWEEN NOW() - INTERVAL 1 DAY AND NOW() GROUP BY round(UNIX_TIMESTAMP(TIME) DIV (60*30))  
ORDER BY `TIME` DESC
